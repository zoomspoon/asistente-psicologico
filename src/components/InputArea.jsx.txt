import React from 'react';

function InputArea({ value, onChange, onSubmit, isLoading }) {
  return (
    <div className="mb-6">
      <textarea
        className="w-full p-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 resize-y min-h-[120px]"
        placeholder="¿Cómo te sientes hoy? Escribe aquí lo que te preocupa..."
        value={value}
        onChange={onChange}
        rows="5"
        disabled={isLoading} // Deshabilita el área mientras la IA está procesando
      ></textarea>
      <button
        onClick={onSubmit}
        className={`w-full py-3 mt-4 rounded-lg text-white font-semibold transition-colors duration-200
          ${isLoading ? 'bg-blue-300 cursor-not-allowed' : 'bg-blue-600 hover:bg-blue-700'}`}
        disabled={isLoading}
      >
        {isLoading ? 'Analizando...' : 'Hablar con el Analista'}
      </button>
    </div>
  );
}

export default InputArea;
